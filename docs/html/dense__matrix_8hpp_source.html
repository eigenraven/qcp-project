274  int total_rows =
275  std::accumulate(mats.begin(), mats.end(), 1,
276  [](int acc, const dmatrix *m) { return acc * m->rows; });
277  int total_cols =
278  std::accumulate(mats.begin(), mats.end(), 1,
279  [](int acc, const dmatrix *m) { return acc * m->cols; });
280  dmatrix kp = dmatrix::zero(total_rows, total_cols);
281  for (int r = 0; r < total_rows; r++) {
282  for (int c = 0; c < total_cols; c++) {
283  // calculate product for the element
284  complex P = 1.0;
285  for (int mi = 0; P != 0.0 && mi < mats.size(); mi++) {
286  const dmatrix &mat = *(mats.at(mi));
287  P *= mat(row_idx[mi], col_idx[mi]);
288  }
289  kp(r, c) = P;

292  col_idx[p]++;
293  if (col_idx[p] >= mats[p]->cols) {
294  col_idx[p] = 0;
295  } else {
296  break;
297  }
298  }
299  }

301  std::fill(col_idx.begin(), col_idx.end(), 0);
