72 template <class M>
73 void QRegisterImpl<M>::applyOperators(
74  gsl::span<std::pair<int, QGate *>> operators, bool disableGrouping,
75  double noise) {
76  std::vector<const M *> matrices;
77  matrices.reserve(this->nqubits);
78  std::vector<bool> covered(nqubits, false);
79  const M *IDmatrix = &ID.matrix<M>();
80  while (operators.size() > 0) {
81  auto nonOverlappingEnd = operators.end();
82  std::fill(covered.begin(), covered.end(), false);
83  int grouped = 0;
84  for (auto it = operators.begin(); it != operators.end(); it++) {
85  auto [qbit, gate] = *it;
86  bool overlaps = std::any_of(covered.begin() + qbit,
87  covered.begin() + qbit + gate->qubits,
88  [](bool b) { return b; });
89  if (overlaps || (disableGrouping && grouped == 1)) {
90  nonOverlappingEnd = it;
91  break;
92  }
93  std::fill(covered.begin() + qbit, covered.begin() + qbit + gate->qubits,true);
94  grouped++;
95  }
96  matrices.clear();
97  for (int i = 0; i < nqubits; i++) {
98  if (!covered.at(i)) {
99  matrices.push_back(IDmatrix);
100  } else {
101  auto [qbit, gate] = *std::find_if(
102  operators.begin(), nonOverlappingEnd,
103  [i](const std::pair<int, QGate *> &a) { return a.first == i; });
104  assert(i == qbit);
105  matrices.push_back(&gate->template matrix<M>());
106  i += gate->qubits - 1;
107  }
108  }
109  M quantumGate = kronecker<M>(gsl::make_span(matrices));
110  state = quantumGate * state;
111  if (nonOverlappingEnd == operators.end()) {
112  break;
113  } else {
114  int removed = nonOverlappingEnd - operators.begin();
115  operators = operators.subspan(removed, operators.size() - removed);
116  }
117  if (noise)
118  applyNoise(noise);
119  }
120 }

123  static std::random_device rd;
124  static std::mt19937 gen(rd() ^ (int)std::time(NULL));
125  static std::uniform_real_distribution<> dis(0.0, 1.0);
126  for (int i = 0; i < nqubits; i++) {
127  if (dis(gen) < noise) {
128  for (int j = 0; j < state.rows; j++) {
129  if (j >> i & 1) {
130  state(j & ~(1 << i), 0) = std::polar(
131  sqrt(pow(std::abs((complex)state(j & ~(1 << i), 0)), 2) +
132  pow(std::abs((complex)state(j, 0)), 2)),
133  std::arg((complex)state(j & ~(1 << i), 0)));

150 template <class M> int QRegisterImpl<M>::measureState() {
151  std::vector<double> norm;
152  norm.reserve(state.rows);
153  for (int row = 0; row < state.rows; row++)
154  norm.push_back(std::norm((complex)state(row, 0)));
155  static std::random_device rd;
156  static std::mt19937 gen(rd() ^ (int)std::time(NULL));
157  static std::uniform_real_distribution<> dis(0.0, 1.0);
158  double random = dis(gen);
159  int state = 0;
160  for (int i = 0; i < norm.size(); i++) {
161  random -= norm[i];
162  if (random <= 0 || i == norm.size() - 1) {
163  state = i;
164  break;
165  }
166  }
